Demo(Add RegisterHelper)

"jQuery CompileHandlebarsTmpl.js" is a jQuery Plugin to use easily "Handlebars.js"

JSON

{
	"Persons": [
		{
			"firstName": "Sato",
			"lastName": "Daiki"
		},
		{
			"firstName": "Hoge",
			"lastName": "Fuga"
		}
	]
}

Template

{{#each Persons}}
<div>{{helper01 firstName}}</div>
<div>{{helper02 lastName}}</div>
{{/each}}

jQuery

	$('.js-compile').compilehandlebarstmpl({
		jsonURL: 'json/test.json',
		getDataType: 'json',
		tmplSrc: '#tmpl',
		helpers: [
			{
				helperName: 'helper01',
				helperFunc: function(name) {
					return name + 'helper01Test';
				}
			},
			{
				helperName: 'helper02',
				helperFunc: function(name) {
					return name + 'helper02Func';
				}
			}
		]
	});
